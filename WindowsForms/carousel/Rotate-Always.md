# Rotate always

The RotateAlways property enables the items in the control to rotate continuously.

{% tabs %}
{% highlight C# %}


this.carousel1.RotateAlways = true;
{% endhighlight %}

{% highlight VB %}


Me.carousel1.RotateAlways = True
{% endhighlight %}

{% endtabs %}
